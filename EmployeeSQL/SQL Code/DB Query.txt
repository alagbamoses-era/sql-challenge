SELECT e.emp_no, e.last_name, e.first_name, e.sex, s.salary
FROM employees AS e
LEFT JOIN salaries AS s
ON e.emp_no = s.salary;

3. CREATE VIEW manager_info AS
SELECT m.dept_no, d.depart_name, m.emp_no, e.last_name, e.first_name
FROM dept_manager AS m
LEFT JOIN departments AS d
ON m.dept_no = d.dept_no
LEFT JOIN dept_emp AS de
ON d.dept_no = de.dept_no
LEFT JOIN employees AS e
ON de.emp_no = e.emp_no;

CREATE VIEW dept_info AS
SELECT de.emp_no, e.last_name, e.first_name, d.depart_name
FROM departments AS d
LEFT JOIN dept_emp AS de
ON d.dept_no = de.dept_no
LEFT JOIN employees AS e
ON de.emp_no = e.emp_no;

CREATE VIEW employees_hercules_b AS

FROM employees

CREATE VIEW employee_sales AS
SELECT de.emp_no, e.last_name, e.first_name, d.depart_name
FROM departments AS d
LEFT JOIN dept_emp AS de
ON d.dept_no = de.dept_no
LEFT JOIN employees AS e
ON de.emp_no = e.emp_no
WHERE depart_name = 'Sales'

CREATE VIEW employee_sales_development AS
SELECT de.emp_no, e.last_name, e.first_name, d.depart_name
FROM departments AS d
LEFT JOIN dept_emp AS de
ON d.dept_no = de.dept_no
LEFT JOIN employees AS e
ON de.emp_no = e.emp_no
WHERE depart_name IN ('Sales', 'Development')
